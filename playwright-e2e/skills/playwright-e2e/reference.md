# Playwright E2E — Setup & Reference

Config patterns, fixture wiring, auth state, and useful commands.
Check the installed version's docs for API drift; the shapes below are stable.

## Install

```bash
npm i -D @playwright/test
npx playwright install              # download browser binaries
npx playwright install --with-deps  # + OS-level deps (useful in CI)
```

## Project layout

```
tests/
  e2e/              # spec files — named by user journey
    checkout-guest.spec.ts
    login.spec.ts
  pages/            # Page Objects — one class per page / major component
    LoginPage.ts
    CheckoutPage.ts
  fixtures/         # Playwright fixture extensions (auth, seeded data, etc.)
    index.ts
    storageState/
      customer.json  # generated by global setup; gitignored
      admin.json
playwright.config.ts
global.setup.ts     # runs once before all tests (creates storage states)
```

## Config — `playwright.config.ts`

```ts
import { defineConfig, devices } from '@playwright/test';

export default defineConfig({
  testDir: './tests/e2e',
  fullyParallel: true,
  retries: process.env.CI ? 1 : 0,   // 1 retry on CI only — fix flakes, don't hide them
  workers: process.env.CI ? 4 : undefined,
  reporter: [['html'], ['github']],
  use: {
    baseURL: process.env.BASE_URL ?? 'http://localhost:3000',
    trace: 'on-first-retry',          // save traces for failed+retried tests
    screenshot: 'only-on-failure',
    video: 'retain-on-failure',
  },
  projects: [
    // Global setup creates auth storage states — runs before test projects
    { name: 'setup', testMatch: /global\.setup\.ts/ },
    {
      name: 'chromium',
      use: {
        ...devices['Desktop Chrome'],
        storageState: 'tests/fixtures/storageState/customer.json',
      },
      dependencies: ['setup'],
    },
  ],
});
```

## Global setup — create auth storage state once

`global.setup.ts`:

```ts
import { test as setup, expect } from '@playwright/test';
import path from 'path';

const CUSTOMER_STATE = path.join('tests/fixtures/storageState', 'customer.json');

setup('authenticate as customer', async ({ page }) => {
  await page.goto('/login');
  await page.getByLabel('Email').fill(process.env.TEST_USER_EMAIL!);
  await page.getByLabel('Password').fill(process.env.TEST_USER_PASSWORD!);
  await page.getByRole('button', { name: 'Sign in' }).click();
  await expect(page).toHaveURL('/dashboard');
  await page.context().storageState({ path: CUSTOMER_STATE });
});
```

Add `tests/fixtures/storageState/*.json` to `.gitignore` — they contain session tokens.

## Fixtures — extend `@playwright/test`

`tests/fixtures/index.ts`:

```ts
import { test as base } from '@playwright/test';
import { LoginPage } from '../pages/LoginPage';
import { CheckoutPage } from '../pages/CheckoutPage';

type AppFixtures = {
  loginPage: LoginPage;
  checkout: CheckoutPage;
};

export const test = base.extend<AppFixtures>({
  loginPage: async ({ page }, use) => {
    await use(new LoginPage(page));
  },
  checkout: async ({ page }, use) => {
    await use(new CheckoutPage(page));
  },
});

export { expect } from '@playwright/test';
```

Import `test` from `../fixtures` (not from `@playwright/test`) in all spec files.

## Page Object pattern

```ts
// tests/pages/CheckoutPage.ts
import { Page, Locator } from '@playwright/test';

export class CheckoutPage {
  readonly page: Page;
  readonly orderConfirmation: Locator;

  constructor(page: Page) {
    this.page = page;
    this.orderConfirmation = page.getByRole('heading', { name: 'Order confirmed' });
  }

  async addToCart(sku: string) {
    await this.page.getByTestId(`add-to-cart-${sku}`).click();
  }

  async placeOrder(): Promise<void> {
    await this.page.getByRole('button', { name: 'Place order' }).click();
    // Returns void; caller asserts the outcome in the spec
  }
}
```

- Locators are defined as class properties (lazy, always fresh — no stale element handles).
- Methods return `void` or the next Page Object; they do **not** contain `expect()` calls.

## Spec pattern

```ts
// tests/e2e/checkout-guest.spec.ts
import { test, expect } from '../fixtures';

test.describe('Guest checkout', () => {
  test('places an order and receives confirmation', async ({ checkout, page }) => {
    await page.goto('/products');
    await checkout.addToCart('SKU-001');
    await page.getByRole('link', { name: 'Cart' }).click();
    await checkout.placeOrder();
    await expect(checkout.orderConfirmation).toBeVisible();
    await expect(page).toHaveURL(/\/orders\/\d+/);
  });
});
```

## API-seeded preconditions

Use the `request` fixture to seed data via the API instead of through the UI when setting
up preconditions — faster and more stable:

```ts
test('displays a previously placed order', async ({ page, request }) => {
  const order = await request.post('/api/orders', {
    data: { items: [{ sku: 'SKU-001', qty: 1 }] },
    headers: { Authorization: 'Bearer ' + process.env.API_TOKEN },
  });
  const { id } = await order.json();

  await page.goto(`/orders/${id}`);
  await expect(page.getByText('SKU-001')).toBeVisible();
});
```

## Network interception

```ts
// Stub a slow or third-party endpoint
await page.route('**/api/recommendations', async route => {
  await route.fulfill({ json: { items: [] } });
});

// Assert a request was made
const responsePromise = page.waitForResponse(
  resp => resp.url().includes('/api/orders') && resp.status() === 201,
);
await checkout.placeOrder();
await responsePromise; // fails fast if the request never happens
```

## Useful commands

```bash
npx playwright test                        # run all tests (headless)
npx playwright test checkout               # run tests matching 'checkout'
npx playwright test --project=chromium     # one browser
npx playwright test --headed               # watch it run
npx playwright test --ui                   # time-travel UI runner
npx playwright test --trace on             # always record traces
npx playwright show-trace trace.zip        # inspect a saved trace
npx playwright test --debug                # launch Inspector, step through
PWDEBUG=1 npx playwright test              # pause on first page.pause()
npx playwright codegen http://localhost:3000  # record a test interactively
npx playwright test --grep @smoke          # run tagged subset
```

## Live-browser exploration with the Playwright MCP server

The `mcp/playwright.json` wires the `@playwright/mcp` server into your AI tool. Use it to:

- Navigate the live app and discover stable locators before writing specs.
- Verify that a user journey completes end-to-end interactively.
- Debug a failing spec by re-running the steps manually with MCP browser control.

The MCP server drives a real browser; it does not replace `npx playwright test` for
automated runs but is invaluable for exploration, locator discovery, and live debugging.

## CI gate example (GitHub Actions)

```yaml
- name: Run Playwright tests
  run: npx playwright test
  env:
    BASE_URL: ${{ vars.STAGING_URL }}
    TEST_USER_EMAIL: ${{ secrets.TEST_USER_EMAIL }}
    TEST_USER_PASSWORD: ${{ secrets.TEST_USER_PASSWORD }}

- name: Upload test artifacts
  if: failure()
  uses: actions/upload-artifact@v4
  with:
    name: playwright-report
    path: |
      playwright-report/
      test-results/
    retention-days: 14
```
